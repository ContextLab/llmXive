% 用于缓存 Data Source，让它和 Usage 自动放在同一行
\newif\ifTaskDataSourcePending
\newcommand{\TaskDataSourceText}{}

\newcommand{\flushtaskdatasource}{%
    \ifTaskDataSourcePending
        \noindent\TaskDataSourceText\par\vspace{0.12em}%
        \TaskDataSourcePendingfalse
    \fi
}

\newcommand{\taskitem}[2]{%
    \ifstrequal{#1}{Data Source}{%
        % 先缓存 Data Source，不立即输出
        \gdef\TaskDataSourceText{\textbf{#1}: #2}%
        \TaskDataSourcePendingtrue
    }{%
        \ifstrequal{#1}{Usage}{%
            % 如果前面有 Data Source，就和 Usage 同一行输出
            \ifTaskDataSourcePending
                \noindent
                \TaskDataSourceText
                \hspace{2em}
                \textbf{#1}: #2
                \par\vspace{0.12em}%
                \TaskDataSourcePendingfalse
            \else
                \noindent\textbf{#1}: #2\par\vspace{0.12em}%
            \fi
        }{%
            % 遇到其他 item 前，先把未输出的 Data Source 刷出来
            \flushtaskdatasource
            \noindent\textbf{#1}: #2\par\vspace{0.12em}%
        }%
    }%
}
