% Side-by-side action+screenshot: one coloured frame holding the trajectory
% text on the left and the captured desktop thumbnail on the right. Used when
% a single GUI/CLI action chunk produces a screenshot we want to surface inline.
%   #1 = image path, #2 = compact caption
% Implementation: VerbatimOut writes the env body to a temp file, then
% \lstinputlisting reads it back as a styled listing inside a minipage; the
% second minipage holds the image. Both live in one tcolorbox frame.
\newenvironment{trajactCLI}[2]{%
  \def\trajaIMG{#1}\def\trajaCAP{#2}%
  \VerbatimEnvironment
  \begin{VerbatimOut}{\jobname.traja.tmp}%
}{%
  \end{VerbatimOut}%
  \begin{tcolorbox}[enhanced, colframe=cligrey, colback=clibg, boxrule=0.6pt,
    arc=2pt, left=3pt, right=3pt, top=2pt, bottom=2pt,
    before skip=3pt, after skip=3pt]
    \begin{minipage}[c]{0.60\linewidth}
      \lstinputlisting[style=HybridChunkBase, backgroundcolor=\color{clibg},
        frame=none, aboveskip=0pt, belowskip=0pt,
        xleftmargin=0pt, framexleftmargin=0pt,
        xrightmargin=0pt, framexrightmargin=0pt]{\jobname.traja.tmp}
    \end{minipage}\hfill
    \begin{minipage}[c]{0.37\linewidth}\centering
      \fcolorbox{cligrey}{white}{\includegraphics[width=0.96\linewidth]{\trajaIMG}}\\[1pt]
      {\scriptsize\color{cligrey}\texttt{\trajaCAP}}
    \end{minipage}
  \end{tcolorbox}%
}
