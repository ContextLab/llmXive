\definecolor{softblue}{RGB}{90, 140, 200}
% Higher-is-better heatmap.


% Higher is better.
\providecommand{\heatb}[3]{}
\renewcommand{\heatb}[3]{%
  \pgfmathsetmacro{\hmdenom}{#3-#2}%
  \pgfmathparse{abs(\hmdenom)<0.00001}%
  \ifnum\pgfmathresult=1\relax
    \def\hmratio{0}%
  \else
    \pgfmathsetmacro{\hmratio}{min(1,max(0,(#1-#2)/\hmdenom))}%
  \fi
  \pgfmathtruncatemacro{\hmshade}{round(10 + 35*\hmratio)}%
  \edef\hmcolor{softblue!\hmshade!white}%
  \expandafter\cellcolor\expandafter{\hmcolor}#1%
}

% Lower is better. Same color scale, reversed direction.
\providecommand{\heatbd}[3]{}
\renewcommand{\heatbd}[3]{%
  \pgfmathsetmacro{\hmdenom}{#3-#2}%
  \pgfmathparse{abs(\hmdenom)<0.00001}%
  \ifnum\pgfmathresult=1\relax
    \def\hmratio{0}%
  \else
    \pgfmathsetmacro{\hmratio}{min(1,max(0,(#3-#1)/\hmdenom))}%
  \fi
  \pgfmathtruncatemacro{\hmshade}{round(10 + 35*\hmratio)}%
  \edef\hmcolor{softblue!\hmshade!white}%
  \expandafter\cellcolor\expandafter{\hmcolor}#1%
}
