\begin{tcolorbox}[
    title=#1,
    colback=casebg, 
    colframe=casetitlebg,
    rounded corners,
    sharp corners=northeast,
    sharp corners=southwest,
    width=1.00\linewidth,
    boxsep=2pt,

    breakable,

}
