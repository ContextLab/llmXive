\newcommand{\colorednumber}[1]{%

    \pgfmathsetmacro{\safecl}{max(0, min(100, \myratio))}%
    \edef\temp{\noexpand\cellcolor{red!\safecl!green}}\temp
    \pgfmathparse{#1 > 450 ? "white" : "black"}%
    \color{\pgfmathresult}#1%
}
